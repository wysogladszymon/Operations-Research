def calculateT0(tc, tm, tp):
  return (tc + 4 * tm + tp) / 6

def calculateVariancy(tp, tc):
  return (tp - tc) ** 2 / 36

def distribuant(X, mu=0, sigma=1):
  return stats.norm.cdf(X, mu, sigma)

def inverseDistribuant(P, mu=0, sigma=1):
  return stats.norm.ppf(P, mu, sigma)

class Node:
  def __init__(self, name, tc, tm, tp):
    self.name = name
    self.tc = tc
    self.tm = tm
    self.tp = tp
    self.t0 = calculateT0(tc, tm, tp)
    self.variancy = calculateVariancy(tp, tc)
    self.predecessors = []
    self.neighbors = []
    self.z = 0
    #earliest and latest start and finish times
    self.es = 0
    self.ef = 0
    self.ls = float('inf')
    self.lf = float('inf')
    
  def addPredecessor(self, node):
    self.predecessors.append(node)
    
  def addNeighbor(self, node):
    self.neighbors.append(node)
    
  def __repr__(self):
    return f'{self.name}'
  
  def printStartFinish(self):
    print(f'{self.name}: {self.es}/{self.ef}/{self.ls}/{self.lf}')

class Graph:
  def __init__(self, tasks):
    self.tasks = tasks
    self.start = tasks[0]
    self.end = tasks[-1]
    self.criticalPath = []
    self.completion = None
    
  def computeEarliest(self):
    # przechodzimy po wszystkich wierzchokach "warstwami" czyli robimy bfs,w ktrym obliczamy najwczeniejszy czas rozpoczcia i zakoczenia
    q = deque()
    q.append(self.start)
    while q:
      node : "Node"= q.popleft()
      # obliczamy najwczeniejszy czas zakoczenia wierzchoka jako maksimum z najwczeniejszych czasw zakoczenia jego poprzednikw (poniewa musz one si zakoczy zanim zacznie si nowe zadanie)
      m = 0
      for prev in node.predecessors:
        if prev.ef > m:
          m = prev.ef
      # obliczone maksimum zapisujemy jako najwczeniejszy czas rozpoczcia i na bazie przewidywanego czasu trwania zadania obliczamy najwczeniejszy czas zakoczenia
      node.es = m
      node.ef = m + node.t0
      # dodajemy do kolejki ssiadw wierzchoka  
      for neighbor in node.neighbors:
        q.append(neighbor)
      
      # jeeli dotarlimy do koca sieci, to zapisujemy czas zakoczenia jako czas zakoczenia sieci
      if node == self.end:
        self.completion = node.ef
        return self.completion
        
  def computeLatest(self):
    # jest to analogiczny algorytm do obliczania najwczeniejszych czasw, ale tym razem obliczamy najpniejsze czasy rozpoczcia i zakoczenia idc od koca sieci
    q = deque()
    node = self.end
    node.z = 0
    node.lf = node.ef
    node.ls = node.es
    for prev in self.end.predecessors:
      q.append(prev)
    while q:
      node = q.popleft()
      m = self.completion
      for neighbor in node.neighbors:
        if neighbor.ls < m:
          m = neighbor.ls
      node.lf = m
      node.ls = m - node.t0
      node.z = (node.ls - node.es) 
      for prev in node.predecessors:
        q.append(prev)
    
  def PERT(self):
    # algorytm PERT wywouje obliczenie najwczeniejszych i najpniejszych czasw rozpoczcia i zakoczenia
    self.computeEarliest()
    self.computeLatest()
    # a nastpnie wyznacza ciek krytyczn algorytmem dfs, suzkajc wierzchokw, ktrych moliwe przesunicie jest rwne 0
    node = self.start
    path = []
    def dfs(node):
      nonlocal path
      # jezeli dotarlimy do koca cieki, to sprawdzamy czy jest ona dusza od dotychczasowej cieki krytycznej i j zapisujemy
      if node == self.end:
        path.append(node)
        if len(path) > len(self.criticalPath):
          self.criticalPath = path.copy()
        return
      # jeeli wierzchoek ma moliwe przesunicie rwne 0, to dodajemy go do cieki i szukamy w jego ssiedztwie
      if node.z == 0:
        path.append(node)
        for neighbor in node.neighbors:
          dfs(neighbor)
        path.pop()
   
    dfs(self.start)
    return self.criticalPath
  
  def totalVariance(self):
    # oblicza wariancj cakowit, jako sum wariancji wierzchokw w ciece krytycznej
    return sum([node.variancy for node in self.criticalPath])
  
  def drawGanttsChart(self):
    # Funkcja rysujca wykres Gantta za pomoc biblioteki matplotlib.pyplot
    plt.title('Gantt chart')
    plt.xlim(0, self.completion + 1)
    # rysowanie zada
    for node in self.tasks[::-1]:
      for neighbor in node.neighbors:
        # dorysowanie przerywanej linii, okrelajcej moliwe przesunicie zadania oraz zalenoci midzy zadaniami
        plt.plot([node.ef, neighbor.es], [node.name, node.name], 'r--', linewidth=1)
        plt.plot([neighbor.es, neighbor.es], [node.name, neighbor.name], 'r--', linewidth=1)
        # dorysowanie linii kocowej zadania
      plt.plot([node.es, node.ef], [node.name, node.name], color='skyblue', linewidth=5)
    plt.grid(True)
    plt.show()
    
    
  def computeTime(self, probability):
    return inverseDistribuant(probability, self.completion, np.sqrt(self.totalVariance()))

def main():
  # ustalenie czynosci i ich czasu trwania
  tasks : List["Node"]= [
      Node('A', 13, 14, 15),
      Node('B', 5, 10, 15),
      Node('C', 7, 10, 19),
      Node('D', 2, 2, 2),
      Node('E', 10, 10, 10),
      Node('F', 20, 21, 22),
      Node('G', 4, 16, 16),
      Node('H', 5, 20, 23),
      Node('I', 5, 8, 11),
      Node('J', 12, 12, 12)
    ]
  #stworzenie grafu
  graph = Graph(tasks)
  
  # dodanie zaleznosci
  edges = [(0,1), (0,2), (0,3), (1,4), (2,5),(2,6),(3,6),(4,8), (5,7), (5,8), (6,7), (8,9), (7,9)]
  
  #dodanie krawdzi do struktury cieki
  for prev, next in edges:
    tasks[prev].addNeighbor(tasks[next])
    tasks[next].addPredecessor(tasks[prev])
  
  # Obliczenia uywajc struktury i funkcji opisanych wczeniej
  crit = graph.PERT()
  print("cieka krytyczna:", crit)
  
  Probability = 0.9
  time = graph.computeTime(Probability)
  print(f'Czas trwania dla prawdopodobiestwa {Probability} wynosi {time.round()}')
  graph.drawGanttsChart()

Pomoce i przemylenia do oblicze zastosowanych w zadaniach

Do obliczenia prawdopodobiestwa naley nasz dystrybuant przeksztaci do dystrybuanty rozkadu normalnego za pomoc wzoru:

\begin{equation} F(X) = \Phi( \frac{X - \mu}{\sigma} ) \end{equation}

gdzie,

$X = t_d$ czas dyrektywy,

$\mu = t_r$ to czas modelowy ukoczenia projektu,

$\sigma = \sigma_c$ to odchylenie standardowe.

Zatem $ t_d $ jest dane wzorem: \begin{equation} t_d = X \cdot \sigma_c + t_r \end{equation} Aby prawdopodobiestwo wynosio 0.9 to, odczytujc z tablic: \begin{equation} \Phi(1.28) = 0.8997 \approx 0.9 \end{equation} Zatem \begin{equation} t_d = 1.28 \cdot \sigma_c + t_r \end{equation}
